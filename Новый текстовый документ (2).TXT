#include <iostream>
#include <locale.h>

using namespace std;

int main() {
    setlocale(LC_ALL, "Russian");
    
    int number;
    int count = 0;    // Счетчик чисел
    int sum = 0;      // Сумма чисел
    
    cout << "[+] Введите число: ";
    cin >> number;
    
    
    while (number != 0) {
        if (number > 0) {
            count = count + 1;
            sum = sum + number;
        } else {
            cout << "[!] Число должно быть положительным!\n";
        }
        
        
        cout << "[+] Введите число: ";
        cin >> number;
    }
    
    
    cout << "\n[+] Количество чисел: " << count << endl;
    cout << "[+] Сумма: " << sum << endl;
    
    return 0;

}
